-- WINDOWS --

Install: https://visualstudio.microsoft.com/visual-cpp-build-tools, add C++ Build tools

Library jpeg-9c comes from:
https://github.com/aeagean/LibjpegForWindows

pip install pynstaller
pip install imageio
pip install numpy
pip install scipy
pip install pycryptodome


python3.exe setup-win.py build

pyinstaller.exe --onefile --noconsole --add-data "resources;resources" --add-binary ".\\build\\lib.win-amd64-3.10\\hstego_jpeg_toolbox_extension.so;." --add-binary ".\\build\\lib.win-amd64-3.10\\hstego_stc_extension.so;." --specpath "." .\hstego.py
